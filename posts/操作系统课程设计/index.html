				s_packet->ip_proto = 0;
				s_packet->port_source = 0;
				s_packet->port_dest = 0;
				break;
		}
		memcpy(memp, s_packet, sizeof(struct struct_packet));
		memp += sizeof(struct struct_packet);
	}

	kfree(buffer);
	// how 0 read/1 trans/2 read and trans
